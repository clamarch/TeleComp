	function checkAndFormatAddress(address, elemID, control, mustBePhysical) {
		var pa = address.Line1 + ", ";
		if (address.Line2 != "") { pa = pa + address.Line2 + ", "; }
		if (address.Line3 != "") { pa = pa + address.Line3 + ", "; }

			document.getElementById(elemID).value = "";
			control.reset();
			control.message(poboxmessage, eopts);
			return false;
		}
		return true;

	function getCookie(c_name) {
		if (document.cookie.length > 0) {
			c_start = document.cookie.indexOf(c_name + "=");
			if (c_start != -1) {
				c_start = c_start + c_name.length + 1;
				c_end = document.cookie.indexOf(";", c_start);
				if (c_end == -1) {
					c_end = document.cookie.length;
				}
				return unescape(document.cookie.substring(c_start, c_end));
			}
		}
		return "";
	}

	//--Read  cookie //


	//create cookie //

	function createCookie(name, value, days) {
		var expires;
		if (days) {
			var date = new Date();
			date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
			expires = "; expires=" + date.toGMTString();
		}
		else {
			expires = "";
		}
		document.cookie = name + "=" + value + expires + "; domain=.distributel.ca;path=/";
	}

function movestep(toredirect) {

  if($("input[name='mod']:checked"). val()=='byod'){
    var slno="";
      var mac=$("#mac").val();
    var slno=$("#slno").val();

    var reg=/^(?=.*[0-9])(?=.*[a-fA-F])([a-fA-F0-9]+)$/;
    jQuery('span.error').remove();
    jQuery('.input-holder').removeClass('error');
    var $error = jQuery('<span class="error"></span>');

    if($('#slno').val()!=""){
      if(slno.indexOf('-')!=-1){
        $('#slno').parent().addClass('error').append($error.clone().text('Please do not include – (Dash) in the SN if there is one.'));
        $('#slno').focus();
        jQuery('span.error').show();
        $('html,body').scrollTop($('.bod.anchor').offset().top);
        return false;
      }
    }


    if($('#mac').val()==""){
      $('#mac').parent().addClass('error').append($error.clone().text('Please enter your Mac Address.'));
      $('#mac').focus();
      jQuery('span.error').show();
      $('html,body').scrollTop($('.bod.anchor').offset().top);
      return false;

    }
    else if($('#mac').val().match(reg)){
     $.ajax({
          type: 'GET',
          url: "https://www.distributel.ca" + '/wp-content/themes/distributel/ajx.php?type=byondmac&mac='+mac+'&slno='+slno,
          success: function(response) {
            createCookie('internet', '1', 30);
            window.location.href = toredirect;

        }
     });

    }
    else{
      $('#mac').parent().addClass('error').append($error.clone().text('Mac Address seems invalid.'));
      $('#mac').focus();
      jQuery('span.error').show();
      $('html,body').scrollTop($('.bod.anchor').offset().top);
      return false;
    }

  }
  else{
     createCookie('internet', '1', 30);
     window.location.href = toredirect;
  }

}
